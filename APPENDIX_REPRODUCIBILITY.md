            direction = action.get('parameters', '').get('direction', '')
            if direction not in ['atas', 'bawah', 'kiri', 'kanan', 'depan', 'belakang']:
                print(f"arah {direction} tidak valid")
                continue

            (x, y, z, r, j1, j2, j3, j4) = device.pose()
            offsets = {

                'bawah': (0, 0, -30),
                'kiri': (0, 40, 0),
                'kanan': (0, -40, 0),
                'depan': (40, 0, 0),
                'belakang': (-30, 0, 0)
            }
            dx, dy, dz = offsets[direction]
            device.move_to(x+dx, y+dy, z+dz, 0, mode=0, wait=True)
        
        elif command == 'move_to':
            parameters = action.get('parameters', '')
            x = parameters.get('x', '')
            y = parameters.get('y', '')
            z = parameters.get('z', '')
            print(f"pergi ke koordinat x={x}, y={y}, z={z}")
            
            # Suction height constraint: if gripping, add 35mm to Z
            if suction_status == False:
                device.move_to(x, y, z, 0, mode=0, wait=True)
            else:
                device.move_to(x, y, z+35, 0, mode=0, wait=True)
        
        elif command == 'suction_cup':
            action_status = action.get('parameters', '').get('action', '')
            if action_status == 'on':
                suction_status = True
                device.suck(True)
            elif action_status == 'off':
                suction_status = False
                device.suck(False)

        elif command == 'err_msg':
            # LLM reports it cannot complete task
            msg = action.get('parameters', '').get('msg', '')
            print(f"LLM Error: {msg}")
        
        else:
            print(f"command : {command} tidak valid")

## C. Perception-to-Coordinate Interface

### C.1 Complete Object Detection System

```python
"""
Perception-to-Coordinate Interface
Converts camera-based object detection to robot workspace coordinates
"""

import cv2
import numpy as np
from typing import List, Tuple, NamedTuple, Optional
from dataclasses import dataclass
from enum import Enum

# ============================================================================
# DATA STRUCTURES
# ============================================================================

class ObjectColor(Enum):
    """Detected object colors"""
    BLUE = "biru"
    ORANGE = "jingga"
    YELLOW = "kuning"

@dataclass
class CameraCalibration:
    """Camera calibration parameters"""
    # Camera coordinate reference points
    camera_top_left: Tuple[int, int]
    camera_bottom_right: Tuple[int, int]
    
    # Corresponding robot workspace coordinates
    robot_top_left: Tuple[float, float]
    robot_bottom_right: Tuple[float, float]
    
    @staticmethod
    def from_config() -> 'CameraCalibration':
        """Load calibration from configuration"""
        return CameraCalibration(
            camera_top_left=(270, 132),
            camera_bottom_right=(142, 338),
            robot_top_left=(294, 122),
            robot_bottom_right=(160, -124)
        )

@dataclass
class BoundingBox:
    """Bounding box in camera coordinates"""
    x: int          # Top-left x
    y: int          # Top-left y
    width: int      # Box width
    height: int     # Box height
    
    @property
    def center_x(self) -> int:
        """Center x coordinate"""
        return self.x + self.width // 2
    
    @property
    def center_y(self) -> int:
        """Center y coordinate"""
        return self.y + self.height // 2
    
    @property
    def area(self) -> int:
        """Bounding box area"""
        return self.width * self.height

@dataclass
class DetectedObject:
    """Detected object with camera and world coordinates"""
    color: ObjectColor
    camera_bbox: BoundingBox
    camera_coords: Tuple[int, int]      # (x, y) in camera frame
    world_coords: Tuple[float, float]   # (x, y) in robot workspace
    confidence: float = 1.0             # Detection confidence
    z_coordinate: float = -50.0         # Default Z (height above surface)
    
    @property
    def action_plan_coords(self) -> str:
        """Format coordinates for action plan"""
        return f"{self.color.value}: ({self.world_coords[0]}, {self.world_coords[1]}, {self.z_coordinate})"

# ============================================================================
# LINEAR INTERPOLATION
# ============================================================================

class LinearInterpolator:
    """Linear interpolation for coordinate transformation"""
    
    def __init__(self, calibration: CameraCalibration):
        self.cal = calibration
        self._validate_calibration()
    
    def _validate_calibration(self):
        """Validate calibration parameters"""
        # Camera points should not be identical
        if self.cal.camera_top_left == self.cal.camera_bottom_right:
            raise ValueError("Camera calibration points are identical")
        
        # Robot points should not be identical
        if self.cal.robot_top_left == self.cal.robot_bottom_right:
            raise ValueError("Robot calibration points are identical")
    
    def camera_to_world(self, camera_x: int, camera_y: int) -> Tuple[float, float]:
        """
        Convert camera coordinates to world coordinates using linear interpolation
        
        Mathematical formulation:
        For each axis independently, apply linear interpolation:
        world_value = world_min + (world_max - world_min) * 
                     ((camera_value - camera_min) / (camera_max - camera_min))
        
        Args:
            camera_x: X coordinate in camera frame (pixels)
            camera_y: Y coordinate in camera frame (pixels)
        
        Returns:
            Tuple of (world_x, world_y) in robot workspace coordinates (mm)
        """
        # Extract calibration points
        cam_x_top_left, cam_y_top_left = self.cal.camera_top_left
        cam_x_bottom_right, cam_y_bottom_right = self.cal.camera_bottom_right
        
        robot_x_top_left, robot_y_top_left = self.cal.robot_top_left
        robot_x_bottom_right, robot_y_bottom_right = self.cal.robot_bottom_right
        
        # Linear interpolation for X
        world_x = self._interpolate(
            camera_x,
            cam_x_top_left,
            cam_x_bottom_right,
            robot_x_top_left,
            robot_x_bottom_right
        )
        
        # Linear interpolation for Y
        world_y = self._interpolate(
            camera_y,
            cam_y_top_left,
            cam_y_bottom_right,
            robot_y_top_left,
            robot_y_bottom_right
        )
        
        return (round(world_x, 2), round(world_y, 2))
    
    @staticmethod
    def _interpolate(value: float, x1: float, x2: float, y1: float, y2: float) -> float:
        """
        Linear interpolation formula:
        y = y1 + (y2 - y1) * ((x - x1) / (x2 - x1))
        
        Args:
            value: Input value to interpolate
            x1: First x reference point
            x2: Second x reference point
            y1: First y value at x1
            y2: Second y value at x2
        
        Returns:
            Interpolated y value
        """
        if x1 == x2:
            raise ValueError("Reference points are identical")
        
        return y1 + (y2 - y1) * ((value - x1) / (x2 - x1))
    
    def batch_convert(self, camera_coords: List[Tuple[int, int]]) -> List[Tuple[float, float]]:
        """Convert multiple points at once"""
        return [self.camera_to_world(x, y) for x, y in camera_coords]

# ============================================================================
# COLOR DETECTION
# ============================================================================

class ColorRange(NamedTuple):
    """HSV color range boundaries"""
    h_lower: int
    s_lower: int
    v_lower: int
    h_upper: int
    s_upper: int
    v_upper: int
    
    @property
    def lower(self) -> np.ndarray:
        """Lower bound array"""
        return np.array([self.h_lower, self.s_lower, self.v_lower], dtype=np.uint8)
    
    @property
    def upper(self) -> np.ndarray:
        """Upper bound array"""
        return np.array([self.h_upper, self.s_upper, self.v_upper], dtype=np.uint8)

class ColorDetector:
    """HSV-based color detection for robot perception"""
    
    # Predefined color ranges (HSV color space)
    COLOR_RANGES = {
        ObjectColor.BLUE: ColorRange(
            h_lower=100, s_lower=50, v_lower=100,
            h_upper=140, s_upper=255, v_upper=255
        ),
        ObjectColor.ORANGE: ColorRange(
            h_lower=5, s_lower=50, v_lower=50,
            h_upper=25, s_upper=255, v_upper=255
        ),
        ObjectColor.YELLOW: ColorRange(
            h_lower=20, s_lower=90, v_lower=100,
            h_upper=40, s_upper=255, v_upper=255
        )
    }
    
    # Size constraints (pixels²)
    MIN_OBJECT_SIZE = 100
    MAX_OBJECT_SIZE = 5000
    
    def __init__(self):
        self.frame_history = []
        self.last_detections = {}
    
    def detect_in_frame(self, frame: np.ndarray, color: ObjectColor) -> Optional[BoundingBox]:
        """
        Detect objects of specific color in frame
        
        Algorithm:
        1. Convert BGR to HSV color space (better for color detection)
        2. Create binary mask using color range
        3. Find contours in mask
        4. Filter contours by size constraints
        5. Return largest contour's bounding box
        
        Args:
            frame: BGR image frame
            color: Color to detect
        
        Returns:
            BoundingBox if found, None otherwise
        """
        # Convert to HSV
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # Get color range
        color_range = self.COLOR_RANGES[color]
        
        # Create mask
        mask = cv2.inRange(hsv, color_range.lower, color_range.upper)
        
        # Apply morphological operations for noise reduction
        kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        
        # Find contours
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if not contours:
            return None
        
        # Filter contours by size and find largest
        valid_contours = [
            cnt for cnt in contours
            if self.MIN_OBJECT_SIZE < cv2.contourArea(cnt) < self.MAX_OBJECT_SIZE
        ]
        
        if not valid_contours:
            return None
        
        # Get largest contour
        largest = max(valid_contours, key=cv2.contourArea)
        x, y, w, h = cv2.boundingRect(largest)
        
        return BoundingBox(x=x, y=y, width=w, height=h)
    
    def detect_all_colors(self, frame: np.ndarray) -> List[BoundingBox]:
        """Detect all target colors"""
        results = []
        for color in ObjectColor:
            bbox = self.detect_in_frame(frame, color)
            if bbox:
                results.append((color, bbox))
        return results

# ============================================================================
# PERCEPTION PIPELINE
# ============================================================================

class PerceptionPipeline:
    """Complete perception-to-coordinate conversion pipeline"""
    
    def __init__(self, calibration: Optional[CameraCalibration] = None):
        self.calibration = calibration or CameraCalibration.from_config()
        self.interpolator = LinearInterpolator(self.calibration)
        self.detector = ColorDetector()
        self.detected_objects: List[DetectedObject] = []
    
    def process_frame(self, frame: np.ndarray) -> List[DetectedObject]:
        """
        Complete perception pipeline
        
        Pipeline stages:
        1. Object detection (color-based)
        2. Bounding box to center point conversion
        3. Camera to world coordinate transformation
        4. Overlap detection and filtering
        5. Confidence filtering
        
        Args:
            frame: Input frame from camera
        
        Returns:
            List of detected objects with world coordinates
        """
        detections = self.detector.detect_all_colors(frame)
        
        if not detections:
            self.detected_objects = []
            return []
        
        # Convert to DetectedObject format
        objects = []
        for color, bbox in detections:
            # Get center point
            camera_x = bbox.center_x
            camera_y = bbox.center_y
            
            # Convert to world coordinates
            world_x, world_y = self.interpolator.camera_to_world(camera_x, camera_y)
            
            # Create detected object
            obj = DetectedObject(
                color=color,
                camera_bbox=bbox,
                camera_coords=(camera_x, camera_y),
                world_coords=(world_x, world_y),
                confidence=1.0,
                z_coordinate=-50.0
            )
            
            objects.append(obj)
        
        # Filter overlapping detections
        objects = self._filter_overlaps(objects)
        
        self.detected_objects = objects
        return objects
    
    def _filter_overlaps(self, objects: List[DetectedObject]) -> List[DetectedObject]:
        """Remove overlapping bounding boxes, keep largest"""
        if len(objects) <= 1:
            return objects
        
        # Sort by area (largest first)
        sorted_objects = sorted(objects, key=lambda o: o.camera_bbox.area, reverse=True)
        
        # Keep non-overlapping
        filtered = []
        for obj in sorted_objects:
            if not self._overlaps_with_any(obj, filtered):
                filtered.append(obj)
        
        return filtered
    
    @staticmethod
    def _overlaps_with_any(obj: DetectedObject, others: List[DetectedObject]) -> bool:
        """Check if object overlaps with any in list"""
        for other in others:
            if PerceptionPipeline._bboxes_overlap(obj.camera_bbox, other.camera_bbox):
                return True
        return False
    
    @staticmethod
    def _bboxes_overlap(bbox1: BoundingBox, bbox2: BoundingBox) -> bool:
        """Check if two bounding boxes overlap (AABB collision)"""
        return (bbox1.x < bbox2.x + bbox2.width and
                bbox1.x + bbox1.width > bbox2.x and
                bbox1.y < bbox2.y + bbox2.height and
                bbox1.y + bbox1.height > bbox2.y)
    
    def get_context_string(self) -> str:
        """Format detected objects as context string for LLM"""
        if not self.detected_objects:
            return "No objects detected in workspace"
        
        lines = ["Objek dan Koordinat yang Tersedia:"]
        for i, obj in enumerate(self.detected_objects, 1):
            lines.append(f"{i}. Balok {obj.color.value} = ({obj.world_coords[0]}, {obj.world_coords[1]}, {obj.z_coordinate})")
        
        return "\n".join(lines)
    
    def visualize_detections(self, frame: np.ndarray) -> np.ndarray:
        """Draw detected objects on frame"""
        output = frame.copy()
        
        for obj in self.detected_objects:
            bbox = obj.camera_bbox
            
            # Draw rectangle
            cv2.rectangle(output,
                         (bbox.x, bbox.y),
                         (bbox.x + bbox.width, bbox.y + bbox.height),
                         (0, 255, 0), 2)
            
            # Draw label
            label = f"{obj.color.value} ({obj.world_coords[0]}, {obj.world_coords[1]})"
            cv2.putText(output, label,
                       (bbox.x, bbox.y - 10),
                       cv2.FONT_HERSHEY_SIMPLEX,
                       0.5, (0, 255, 0), 2)
        
        return output

def detect_blocks(frame, min_size=100, max_size=5000):
    """Detect colored blocks using HSV segmentation"""
    hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
    
    # HSV color ranges used in production
    colors_range = {
        'biru': {
            'lower': np.array([100, 50, 100]),
            'upper': np.array([140, 255, 255])
        },
        'oranye': {
            'lower': np.array([5, 50, 50]),
            'upper': np.array([25, 255, 255])
        },
        'kuning': {
            'lower': np.array([20, 90, 100]),
            'upper': np.array([40, 255, 255])
        }
    }
    
    detected_coordinates = []
    
    for color_name, color_range in colors_range.items():
        mask = cv2.inRange(hsv, color_range['lower'], color_range['upper'])
        kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)
        
        for contour in contours:
            area = cv2.contourArea(contour)
            if min_size <= area <= max_size:
                M = cv2.moments(contour)
                if M['m00'] > 0:
                    camera_x = int(M['m10'] / M['m00'])
                    camera_y = int(M['m01'] / M['m00'])
                    detected_coordinates.append((color_name, (camera_x, camera_y)))
    
    # Remove overlapping detections
    final_detections = []
    for color_name, (camera_x, camera_y) in detected_coordinates:
        is_overlapping = False
        for existing_color, (existing_x, existing_y) in final_detections:
            distance = ((camera_x - existing_x)**2 + (camera_y - existing_y)**2)**0.5
            if distance < 30:
                is_overlapping = True
                break
        if not is_overlapping:
            final_detections.append((color_name, (camera_x, camera_y)))
    
    return final_detections
```

```python
def maps_to_real(maps_x, maps_y):
    """Linear interpolation for camera-to-world coordinate conversion"""
    
    # Calibration points
    camera_top_left = (270, 132)
    camera_bottom_right = (142, 338)
    world_top_left = (294, 122)
    world_bottom_right = (160, -124)
    
    # Extract min/max values
    camera_min_x = camera_bottom_right[0]  # 142
    camera_max_x = camera_top_left[0]      # 270
    camera_min_y = camera_top_left[1]      # 132
    camera_max_y = camera_bottom_right[1]  # 338
    
    world_min_x = world_bottom_right[0]    # 160
    world_max_x = world_top_left[0]        # 294
    world_min_y = world_bottom_right[1]    # -124
    world_max_y = world_top_left[1]        # 122
    
    # Linear interpolation
    real_x = world_min_x + (world_max_x - world_min_x) * \
             ((maps_x - camera_min_x) / (camera_max_x - camera_min_x))
    
    real_y = world_min_y + (world_max_y - world_min_y) * \
             ((maps_y - camera_min_y) / (camera_max_y - camera_min_y))
    
    return real_x, real_y
```

**Calibration Matrix:**

| Point | Camera Coords (pixels) | Robot Coords (mm) |
|-------|------------------------|-------------------|
| Top-Left (Workspace) | (270, 132) | (294, 122) |
| Bottom-Right (Workspace) | (142, 338) | (160, -124) |

**Transformation Formula:**

$$x_{world} = 160 + (294 - 160) \times \frac{x_{cam} - 142}{270 - 142}$$

$$y_{world} = -124 + (122 - (-124)) \times \frac{y_{cam} - 132}{338 - 132}$$

### D.1 Actual Production Implementation

The execution layer converts validated JSON action plans into actual robot movements. The complete implementation is contained in [app.py - RobotExecute2() function (lines 120-171)](implementasi/app.py#L120):

```python
def RobotExecute2(data):
    """
    Execute JSON action plan on Dobot Magician robot
    
    Args:
        data: List of action dictionaries from validated JSON
              Each action has 'command' and 'parameters' keys
    
    Robot workspace constraints (implicit):
        X: -400 to 400 mm
        Y: -400 to 400 mm
        Z: -150 to 200 mm
    """
    # Initialize serial connection to robot (COM3 at 115200 baud)
    device = pydobot.Dobot(port="COM3", verbose=False)
    suction_status = False
    
    for action in data:
        command = action.get('command', '')
        
        # =====================================================================
        # MOVE COMMAND: Relative movement by fixed offset
        # =====================================================================
        if command == 'move':
            direction = action.get('parameters', '').get('direction', '')
            
            # Validate direction
            valid_directions = ['atas', 'bawah', 'kiri', 'kanan', 'depan', 'belakang']
            if direction not in valid_directions:
                print(f"arah {direction} tidak valid")
                continue
            
            # Get current pose from robot
            (x, y, z, r, j1, j2, j3, j4) = device.pose()
            
            # Direction offsets (in mm)
            # Workspace calibration constraints built-in to offsets
            offsets = {
                'atas': (0, 0, 20),              # Move up
                'bawah': (0, 0, -30),            # Move down
                'kiri': (0, 40, 0),              # Move left
                'kanan': (0, -40, 0),            # Move right
                'depan': (40, 0, 0),             # Move forward
                'belakang': (-30, 0, 0)          # Move backward
            }
            dx, dy, dz = offsets[direction]
            
            # Compute target position
            new_x = x + dx
            new_y = y + dy
            new_z = z + dz
            
            # Apply suction height offset if gripping
            if suction_status:
                new_z = new_z + 35  # Add 35mm when suction active
            
            # Send movement command to robot
            device.move_to(new_x, new_y, new_z, 0, mode=0, wait=True)
            print(f"pergi ke {direction}: x={new_x}, y={new_y}, z={new_z}")
        
        # =====================================================================
        # MOVE_TO COMMAND: Absolute movement to coordinates
        # =====================================================================
        elif command == 'move_to':
            parameters = action.get('parameters', '')
            x = parameters.get('x', '')
            y = parameters.get('y', '')
            z = parameters.get('z', '')
            
            # Apply suction height offset
            if suction_status:
                z = z + 35
            
            print(f"pergi ke koordinat x={x}, y={y}, z={z}")
            device.move_to(x, y, z, 0, mode=0, wait=True)
        
        # =====================================================================
        # SUCTION_CUP COMMAND: End-effector control
        # =====================================================================
        elif command == 'suction_cup':
            action_status = action.get('parameters', '').get('action', '')
            
            if action_status == 'on':
                suction_status = True
                device.suck(True)
                print("penyedot aktif")
            elif action_status == 'off':
                suction_status = False
                device.suck(False)
                print("penyedot non-aktif")
            else:
                print(f"status penyedot {action_status} tidak valid")
        
        # =====================================================================
        # ERR_MSG COMMAND: Error fallback when LLM cannot plan
        # =====================================================================
        elif command == 'err_msg':
            msg = action.get('parameters', '').get('msg', '')
            print(f"LLM Error: {msg}")
        
        # =====================================================================
        # UNKNOWN COMMAND
        # =====================================================================
        else:
            print(f"command : {command} tidak valid")
    
    # Cleanup: return to home position and disable suction
    print(f"kembali ke home")
    device.move_to(250, 0, 0, 0, mode=0, wait=True)
    device.suck(False)
```

### D.2 Workspace Constraints and Kinematic Limits

**Workspace Boundaries** (implicit in algorithm):
- **X range**: -400 to 400 mm (checked implicitly by movement offsets and move_to validation)
- **Y range**: -400 to 400 mm
- **Z range**: -150 to 200 mm
- **Default home position**: (250, 0, 0) with rotation R=0

**Movement Offset Table** (from offset dictionary):

| Direction | Delta X | Delta Y | Delta Z | Notes |
|-----------|---------|---------|---------|-------|
| atas (up) | 0 | 0 | +20 mm | Increase height |
| bawah (down) | 0 | 0 | -30 mm | Decrease height |
| depan (forward) | +40 mm | 0 | 0 | Positive X direction |
| belakang (backward) | -30 mm | 0 | 0 | Negative X direction |
| kiri (left) | 0 | +40 mm | 0 | Positive Y direction |
| kanan (right) | 0 | -40 mm | 0 | Negative Y direction |

**Suction Height Offset**:
- When `suction_status == True`: Add 35 mm to Z coordinate
- Purpose: Prevent dragging gripped objects on surface during movement
- Applied to both `move` and `move_to` commands

### D.5 Error Handling

The execution layer implements basic error handling:

1. **Invalid command type**: Printed error, action skipped
2. **Invalid direction**: Printed error, action skipped
3. **Invalid suction state**: Printed error, action skipped
4. **Serial communication failure**: Exception caught (implicit in pydobot), action fails
5. **Out-of-workspace**: Movement executed (no explicit validation, relies on robot firmware)

class ExecutionContext:
    """Context for action execution including state and constraints"""
    
    def __init__(self, robot_driver, constraints: Optional[WorkspaceConstraints] = None):
        self.robot = robot_driver
        self.constraints = constraints or WorkspaceConstraints()
        self.current_pose = self._get_initial_pose()
        self.suction_status = False
        self.movement_history: List[Dict] = []
        self.state = RobotState.IDLE
    
    def _get_initial_pose(self) -> RobotPose:
        """Get current robot pose"""
        try:
            x, y, z, r, j1, j2, j3, j4 = self.robot.pose()
            return RobotPose(x=x, y=y, z=z, r=r, j1=j1, j2=j2, j3=j3, j4=j4)
        except Exception as e:
            logger.warning(f"Could not get initial pose: {e}")
            return RobotPose(x=250, y=0, z=0, r=0)
    
    def validate_move(self, target_x: float, target_y: float, target_z: float) -> Tuple[bool, str]:
        """Validate a move against kinematic constraints"""
        is_valid, msg = self.constraints.is_point_in_workspace(target_x, target_y, target_z)
        
        if not is_valid:
            return False, f"Target point out of workspace: {msg}"
        
        # Check for self-collision (simplified check)
        distance = self.current_pose.distance_to(RobotPose(
            x=target_x, y=target_y, z=target_z, r=self.current_pose.r
        ))
        
        # If movement is very large, might be unreachable
        if distance > 500:  # Arbitrary threshold
            return True, f"Large movement detected: {distance:.1f}mm (may be unreachable)"
        
        return True, "Move is valid"
    
    def apply_suction_height_offset(self, z: float) -> float:
        """
        Apply height offset when suction is active
        
        When gripping object with suction, increase Z to prevent
        dragging object on surface
        """
        if self.suction_status:
            return z + 35  # 35mm offset
        return z
    
    def update_pose(self, x: float, y: float, z: float, r: float = 0):
        """Update recorded pose after successful movement"""
        self.current_pose = RobotPose(x=x, y=y, z=z, r=r)
        self.movement_history.append({
            'timestamp': time.time(),
            'pose': (x, y, z, r),
            'suction': self.suction_status

class ActionExecutor:
    """Executes validated JSON actions on robot"""
    
    def __init__(self, execution_context: ExecutionContext):
        self.context = execution_context
    
    def execute_action(self, action: Dict) -> bool:
        """
        Execute single action
        
        Args:
            action: Action dict with 'command' and 'parameters'
        
        Returns:
            True if successful, False otherwise
        """
        command = action.get('command')
        
        try:
            if command == 'move':
                return self._execute_move(action)
            elif command == 'move_to':
                return self._execute_move_to(action)
            elif command == 'suction_cup':
                return self._execute_suction(action)
            elif command == 'err_msg':
                return self._execute_error(action)
            else:
                logger.error(f"Unknown command: {command}")
                return False
        
        except Exception as e:
            logger.error(f"Error executing {command}: {e}")
            self.context.state = RobotState.ERROR
            return False
    
    def _execute_move(self, action: Dict) -> bool:
        """Execute relative movement"""
        logger.info(f"Executing MOVE command")
        
        direction = action['parameters'].get('direction')
        
        # Define offsets for each direction
        offsets = {
            'atas': (0, 0, 20),
            'bawah': (0, 0, -30),
            'depan': (40, 0, 0),
            'belakang': (-30, 0, 0),
            'kiri': (0, 40, 0),
            'kanan': (0, -40, 0)
        }
        
        if direction not in offsets:
            logger.error(f"Unknown direction: {direction}")
            return False
        
        dx, dy, dz = offsets[direction]
        target_x = self.context.current_pose.x + dx
        target_y = self.context.current_pose.y + dy
        target_z = self.context.current_pose.z + dz
        
        # Apply suction offset if active
        target_z = self.context.apply_suction_height_offset(target_z)
        
        # Validate
        is_valid, msg = self.context.validate_move(target_x, target_y, target_z)
        if not is_valid:
            logger.error(f"Invalid move: {msg}")
            return False
        
        logger.info(f"  Direction: {direction}")
        logger.info(f"  From: ({self.context.current_pose.x}, {self.context.current_pose.y}, {self.context.current_pose.z})")
        logger.info(f"  To:   ({target_x}, {target_y}, {target_z})")
        
        # Execute movement
        try:
            self.context.state = RobotState.MOVING
            self.context.robot.move_to(target_x, target_y, target_z, 0, mode=0, wait=True)
            self.context.update_pose(target_x, target_y, target_z)
            self.context.state = RobotState.IDLE
            return True
        except Exception as e:
            logger.error(f"Movement failed: {e}")
            return False
    
    def _execute_move_to(self, action: Dict) -> bool:
        """Execute absolute movement to coordinates"""
        logger.info(f"Executing MOVE_TO command")
        
        params = action['parameters']
        target_x = params['x']
        target_y = params['y']
        target_z = params['z']
        
        # Apply suction offset if active
        target_z = self.context.apply_suction_height_offset(target_z)
        
        # Validate
        is_valid, msg = self.context.validate_move(target_x, target_y, target_z)
        if not is_valid:
            logger.error(f"Invalid move_to: {msg}")
            return False
        
        logger.info(f"  From: ({self.context.current_pose.x}, {self.context.current_pose.y}, {self.context.current_pose.z})")
        logger.info(f"  To:   ({target_x}, {target_y}, {target_z})")
        
        # Execute movement
        try:
            self.context.state = RobotState.MOVING
            self.context.robot.move_to(target_x, target_y, target_z, 0, mode=0, wait=True)
            self.context.update_pose(target_x, target_y, target_z)
            self.context.state = RobotState.IDLE
            return True
        except Exception as e:
            logger.error(f"Movement failed: {e}")
            return False
    
    def _execute_suction(self, action: Dict) -> bool:
        """Execute suction cup control"""
        logger.info(f"Executing SUCTION_CUP command")
        
        action_state = action['parameters'].get('action')
        
        try:
            if action_state == 'on':
                self.context.state = RobotState.GRIPPING
                self.context.robot.suck(True)
                self.context.suction_status = True
                logger.info(f"  Suction cup turned ON")
            elif action_state == 'off':
                self.context.state = RobotState.IDLE
                self.context.robot.suck(False)
                self.context.suction_status = False
                logger.info(f"  Suction cup turned OFF")
            else:
                logger.error(f"Unknown suction state: {action_state}")
                return False
            
            return True
        except Exception as e:
            logger.error(f"Suction control failed: {e}")
            return False
    
    def _execute_error(self, action: Dict) -> bool:
        """Handle error message from LLM"""
        msg = action['parameters'].get('msg', 'Unknown error')
        logger.warning(f"LLM reported error: {msg}")
        return True

class ActionPlanExecutor:
    """Executes complete action plans with safety checks"""
    
    def __init__(self, robot_driver, constraints: Optional[WorkspaceConstraints] = None):
        self.context = ExecutionContext(robot_driver, constraints)
        self.executor = ActionExecutor(self.context)
        self.execution_log: List[Dict] = []
    
    def execute_plan(self, action_plan: Dict) -> Tuple[bool, str]:
        """
        Execute complete action plan with error recovery
        
        Args:
            action_plan: Complete action plan with 'actions' array
        
        Returns:
            Tuple of (success: bool, summary: str)
        """
        actions = action_plan.get('actions', [])
        
        if not actions:
            return False, "No actions in plan"
        
        logger.info(f"Starting execution of {len(actions)} actions")
        
        successful = 0
        failed = 0
        
        for idx, action in enumerate(actions):
            logger.info(f"\n[{idx+1}/{len(actions)}] Executing action: {action.get('command')}")
            
            if self.executor.execute_action(action):
                successful += 1
                self.execution_log.append({
                    'index': idx,
                    'action': action,
                    'status': 'success',
                    'timestamp': time.time()
                })
            else:
                failed += 1
                self.execution_log.append({
                    'index': idx,
                    'action': action,
                    'status': 'failed',
                    'timestamp': time.time()
                })
                logger.error(f"Action {idx} failed, stopping execution")
                break
        
        # Return to home position
        try:
            logger.info("\nReturning to home position")
            self.context.robot.move_to(250, 0, 0, 0, mode=0, wait=True)
            self.context.robot.suck(False)
            self.context.suction_status = False
        except Exception as e:
            logger.warning(f"Failed to return to home: {e}")
        
        summary = f"Executed {successful}/{len(actions)} actions successfully"
        success = (failed == 0)
        
        logger.info(f"\n{'='*50}")
        logger.info(f"EXECUTION SUMMARY: {summary}")
        logger.info(f"Status: {'SUCCESS' if success else 'PARTIAL FAILURE'}")
        logger.info(f"{'='*50}\n")
        
        return success, summary
